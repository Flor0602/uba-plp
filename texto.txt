podemos_ganarle_a_thanos :: Universo -> Bool
podemos_ganarle_a_thanos universo = not (está_el_personaje "Thanos" universo) || if not (tiene_thanos_todas_las_gemas universo) 
                                    then ((está_el_personaje "Thor" universo && está_el_objeto "StormBreaker" universo) || 
                                    (está_el_personaje "Wanda" universo && está_el_personaje "Visión" universo && 
                                    en_posesion_de "Visión" (objeto_de_nombre "Gema de la Mente" universo))  && 
                                    wandaMataAVision "Gema de la Mente" "Visión" universo) else False

wandaMataAVision :: String -> String -> Universo -> Bool
wandaMataAVision gema vision universo = fue_destruido (objeto_de_nombre gema universo) && not (está_vivo (personaje_de_nombre vision universo))
